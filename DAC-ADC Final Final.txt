// DAC -> ADC loopback ESP32 (Arduino)
// Añadido: lectura de potenciómetro en ADC_PIN1 (GPIO35)
// y uso del pot para modificar AMPLITUD o FRECUENCIA.

const int DAC_PIN = 25;   // DAC1
const int ADC_PIN = 34;   // ADC1_CH6 (loopback)
const int ADC_PIN1 = 35;  // ADC para potenciómetro (input-only)

int dacValue = 0;

// Para controlar "frecuencia" usamos paso (step). Guardamos signo por separado.
int stepAbs = 5;    // magnitud del paso (1..20 típicamente)
int stepSign = 1;   // 1 = subiendo, -1 = bajando

// Modo: poner true para que el pot controle AMPLITUD.
// poner false para que el pot controle FRECUENCIA (velocidad).
#define MODE_AMPLITUDE true
//#define MODE_AMPLITUDE false  // descomenta según prefieras

void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println("ESP32 DAC -> ADC loopback test");
  Serial.println("GPIO25 (DAC) -> GPIO34 (ADC)");
  Serial.println("GPIO35 (pot) controla " 
    #if MODE_AMPLITUDE
      "AMPLITUD"
    #else
      "FRECUENCIA"
    #endif
  );
  
  // (Opcional) asegurar resolución ADC si tu core lo soporta:
  // analogReadResolution(12); // 0..4095
}

void loop() {
  // ---- leer potenciómetro (ADC_PIN1) ----
  int potRaw = analogRead(ADC_PIN1); // 0 .. 4095
  // ---- MAPEO dependiendo del modo ----
#if MODE_AMPLITUDE
  // Mapear pot a un máximo de DAC (mínimo 10 para no dejarlo en 0)
  int dacMax = map(potRaw, 0, 4095, 10, 255); // 10 .. 255
  // mantener stepAbs constante (puedes ajustar este valor)
  stepAbs = 5;
#else
  // En modo FRECUENCIA, pot controla la velocidad (magnitud del paso)
  // pasos pequeños => onda lenta; pasos grandes => onda rápida
  stepAbs = map(potRaw, 0, 4095, 1, 20); // 1 .. 20
  // amplitud fija a full
  int dacMax = 255;
#endif

  // ---- generar onda triangular entre 0 y dacMax ----
  dacValue += stepSign * stepAbs;

  // comprobar y voltear dirección si alcanzó límites
  if (dacValue >= dacMax) {
    dacValue = dacMax;
    stepSign = -1;
  } else if (dacValue <= 0) {
    dacValue = 0;
    stepSign = 1;
  }

  // escribir DAC (0..255)
  dacWrite(DAC_PIN, dacValue);

  // leer ADC de loopback (0..4095)
  int adcRaw = analogRead(ADC_PIN);

  // Conversión aproximada a voltaje (opcional)
  float vDac = (dacValue / 255.0) * 3.3;
  float vAdc = (adcRaw / 4095.0) * 3.3;
}
